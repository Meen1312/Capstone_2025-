    (df['THRESHOLD_2023'].str.lower() == 'below') &
    (df['THRESHOLD_2024'].str.lower() == 'above')
]

    ['ZIP_CODE', 'ANNUAL_VOLUME_CASES_2024', 'ANNUAL_VOLUME_GALLON_2024']
].dropna()

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Step 1: Load the dataset (ONLY run once to avoid overwriting)
file_path = "sccu_data.csv"
df = pd.read_csv(file_path)
print("✅ Data loaded. Shape:", df.shape)

# Step 2: Define transitioned_customers consistently
transitioned_customers = df[df["THRESHOLD_2023"].notnull()].copy()

# Step 3: Filter valid rows for cost-per-case calculation
valid_cost_data = transitioned_customers[
    (transitioned_customers["DELIVERY_COST_2024"] > 0) &
    (transitioned_customers["ANNUAL_VOLUME_CASES_2024"] > 0)
].copy()

# Step 4: Calculate cost per case and average
valid_cost_data["Cost_Per_Case"] = valid_cost_data["DELIVERY_COST_2024"] / valid_cost_data["ANNUAL_VOLUME_CASES_2024"]
cost_per_case = valid_cost_data["Cost_Per_Case"].mean()
print(f"✅ Actual Average Delivery Cost per Case (2024): ${cost_per_case:.2f}")

# Step 5: Prepare growth_customers by merging volume data
# If growth_customers doesn't already include case volume, merge it in
growth_customers = growth_customers.merge(
    transitioned_customers[["CUSTOMER_NUMBER", "ANNUAL_VOLUME_CASES_2024"]],
    how="left",
    on="CUSTOMER_NUMBER"
)

# Step 6: Rename for clarity
growth_customers.rename(columns={"ANNUAL_VOLUME_CASES_2024": "Total_Ordered_Cases"}, inplace=True)

# Step 7: Group by Cluster and estimate cost
cluster_summary = growth_customers.groupby("Cluster").agg({
    "Total_Ordered_Cases": ["count", "sum", "mean"]
}).reset_index()

cluster_summary.columns = ["Cluster", "Customer_Count", "Total_Cases", "Avg_Cases_Per_Customer"]
cluster_summary["Estimated_Cost"] = cluster_summary["Total_Cases"] * cost_per_case

# Step 8: Display summary
print("\n💰 Estimated Delivery Cost by Cluster (using actual average cost per case):\n")
print(cluster_summary[["Cluster", "Customer_Count", "Total_Cases", "Estimated_Cost"]])

# Step 9: Visualization
plt.figure(figsize=(10, 6))
sns.barplot(data=cluster_summary, x="Cluster", y="Estimated_Cost", palette="mako")
plt.title("Estimated Delivery Cost by Cluster (Real Cost Basis)")
plt.xlabel("Cluster")
plt.ylabel("Estimated Delivery Cost ($)")
plt.tight_layout()
plt.show()

    "DELIVERY_COST_2024": "sum",
